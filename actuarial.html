Python – GLM Poisson

# Python
import pandas as pd, statsmodels.api as sm
X = pd.get_dummies(df[['age_band','wealth_class']], drop_first=True)
X = sm.add_constant(X)
y = df['claims']
model = sm.GLM(y, X, family=sm.families.Poisson()).fit()
print(model.summary())
